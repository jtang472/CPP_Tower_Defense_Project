187  template<typename TData>
188  void DataManager::loadData(game::data::DataType type) {
189  std::string path = mTypeToFileMapping[type];
190  auto &dataMap = mAllData[type];
191 
192  using namespace std;
193  std::ifstream jsonFIle(path);
194  stringstream ss;
195  ss << jsonFIle.rdbuf();
196  string json = ss.str();
197  rapidjson::Document doc;
198  if(!doc.Parse(json.c_str()).HasParseError() && doc.IsArray()) {
199  const rapidjson::Value &array = doc;
200  for(size_t i = 0; i < array.Size(); i++) {
201  const rapidjson::Value& json = array[i];
202  if(array[i].IsObject()) {
203  dataMap.insert(std::make_pair(json["tid"].GetInt(), new TData {json}));
204  }
205  }
206  }
207  }

210  TData* DataManager::get(game::data::DataType type, int tid) {
211  auto iter = mAllData.find(type);
212  if(iter == mAllData.end())
213  return nullptr;
214 
215  DataMap &map = iter->second;
216  auto dataIter = map.find(tid);
217  if(dataIter == map.end())
218  return nullptr;
219 
220  return static_cast<TData*>(dataIter->second.get());
